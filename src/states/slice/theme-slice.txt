import { createSlice } from '@reduxjs/toolkit';

interface Theme {
  theme: string;
}

const initialState: Theme = {
  theme: 'light',
};

const themeSlice = createSlice({
  name: 'theme',
  initialState,
  reducers: {
    setDark(state) {
      state.theme = 'dark'
    },
    setLight(state) {
      state.theme = 'light';
    },
  },
});

export const { setDark, setLight } = themeSlice.actions;

export default themeSlice.reducer;
